class Solution(object):
    def moveZeroes(self, nums):
        """
        :type nums: List[int]
        :rtype: None Do not return anything, modify nums in-place instead.
        """
       
        """
        Do not return anything, modify nums in-place instead.
        """
        temp=[]
        c=0
        for i in nums:
            if(i==0):
                c+=1
            else:
                temp.append(i)
        for i in range(c):
             temp.append(0)
        nums[:]=temp
